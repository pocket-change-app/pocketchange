Input the following queries into playground to test:
query User($userID: ID){
    user(userID: $userID) {
      userID
    }
}

query Business($busID: ID){
    business(busID: $busID) {
      busID
      pocketID
    }
  }
  
query Pocket($pocketID:ID){
  pocket(pocketID:$pocketID) {
    pocketID
    pocketName
    customers
    businesses
    changeRate
    circulatingPoints
  }
}

query loginUser($username:String, $password: String){
  loginUser(username:$username, password: $password) {
    userID
  }
}

query loginBus($busname:String, $password: String){
  loginBus(busname:$busname, password: $password) {
    busID
    pocketID
    busname
  }
}
